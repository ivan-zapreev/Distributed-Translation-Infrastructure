173  asio::io_service::fork_event fork_ev)
174 {
175 #if !defined(ASIO_WINDOWS) \
176  && !defined(ASIO_WINDOWS_RUNTIME) \
177  && !defined(__CYGWIN__)
178  signal_state* state = get_signal_state();
179  static_mutex::scoped_lock lock(state->mutex_);
180 
181  switch (fork_ev)
182  {
183  case asio::io_service::fork_prepare:
184  {
185  int read_descriptor = state->read_descriptor_;
186  state->fork_prepared_ = true;
187  lock.unlock();
188  reactor_.deregister_internal_descriptor(read_descriptor, reactor_data_);
189  }
190  break;
191  case asio::io_service::fork_parent:
192  if (state->fork_prepared_)
193  {
194  int read_descriptor = state->read_descriptor_;
195  state->fork_prepared_ = false;
196  lock.unlock();
197  reactor_.register_internal_descriptor(reactor::read_op,
198  read_descriptor, reactor_data_, new pipe_read_op);
199  }
200  break;
201  case asio::io_service::fork_child:
202  if (state->fork_prepared_)
203  {
204  asio::detail::signal_blocker blocker;
205  close_descriptors();
206  open_descriptors();
207  int read_descriptor = state->read_descriptor_;
208  state->fork_prepared_ = false;
209  lock.unlock();
210  reactor_.register_internal_descriptor(reactor::read_op,
211  read_descriptor, reactor_data_, new pipe_read_op);
212  }
213  break;
214  default:
215  break;
216  }
217 #else // !defined(ASIO_WINDOWS)
218  // && !defined(ASIO_WINDOWS_RUNTIME)
219  // && !defined(__CYGWIN__)

240 asio::error_code signal_set_service::add(
241  signal_set_service::implementation_type& impl,
242  int signal_number, asio::error_code& ec)
243 {
244  // Check that the signal number is valid.
245  if (signal_number < 0 || signal_number >= max_signal_number)
246  {
247  ec = asio::error::invalid_argument;
248  return ec;
249  }
250 
251  signal_state* state = get_signal_state();
252  static_mutex::scoped_lock lock(state->mutex_);
253 
254  // Find the appropriate place to insert the registration.
255  registration** insertion_point = &impl.signals_;
256  registration* next = impl.signals_;
257  while (next && next->signal_number_ < signal_number)
258  {
259  insertion_point = &next->next_in_set_;
260  next = next->next_in_set_;
261  }
262 
263  // Only do something if the signal is not already registered.
264  if (next == 0 || next->signal_number_ != signal_number)
265  {
266  registration* new_registration = new registration;
267 
268 #if defined(ASIO_HAS_SIGNAL) || defined(ASIO_HAS_SIGACTION)
269  // Register for the signal if we're the first.
270  if (state->registration_count_[signal_number] == 0)
271  {
272 # if defined(ASIO_HAS_SIGACTION)
273  using namespace std; // For memset.
274  struct sigaction sa;
275  memset(&sa, 0, sizeof(sa));
276  sa.sa_handler = asio_signal_handler;
277  sigfillset(&sa.sa_mask);
278  if (::sigaction(signal_number, &sa, 0) == -1)
279 # else // defined(ASIO_HAS_SIGACTION)
280  if (::signal(signal_number, asio_signal_handler) == SIG_ERR)
281 # endif // defined(ASIO_HAS_SIGACTION)
282  {
283 # if defined(ASIO_WINDOWS) || defined(__CYGWIN__)
284  ec = asio::error::invalid_argument;
285 # else // defined(ASIO_WINDOWS) || defined(__CYGWIN__)
286  ec = asio::error_code(errno,
287  asio::error::get_system_category());
288 # endif // defined(ASIO_WINDOWS) || defined(__CYGWIN__)
289  delete new_registration;
290  return ec;
291  }
292  }
293 #endif // defined(ASIO_HAS_SIGNAL) || defined(ASIO_HAS_SIGACTION)
294 
295  // Record the new registration in the set.
296  new_registration->signal_number_ = signal_number;
297  new_registration->queue_ = &impl.queue_;
298  new_registration->next_in_set_ = next;
299  *insertion_point = new_registration;
300 
301  // Insert registration into the registration table.
302  new_registration->next_in_table_ = registrations_[signal_number];
303  if (registrations_[signal_number])
304  registrations_[signal_number]->prev_in_table_ = new_registration;
305  registrations_[signal_number] = new_registration;
306 
307  ++state->registration_count_[signal_number];
308  }
309 
310  ec = asio::error_code();
311  return ec;
312 }

314 asio::error_code signal_set_service::remove(
315  signal_set_service::implementation_type& impl,
316  int signal_number, asio::error_code& ec)
317 {
318  // Check that the signal number is valid.
319  if (signal_number < 0 || signal_number >= max_signal_number)
320  {
321  ec = asio::error::invalid_argument;
322  return ec;
323  }
324 
325  signal_state* state = get_signal_state();
326  static_mutex::scoped_lock lock(state->mutex_);
327 
328  // Find the signal number in the list of registrations.
329  registration** deletion_point = &impl.signals_;
330  registration* reg = impl.signals_;
331  while (reg && reg->signal_number_ < signal_number)
332  {
333  deletion_point = &reg->next_in_set_;
334  reg = reg->next_in_set_;
335  }
336 
337  if (reg != 0 && reg->signal_number_ == signal_number)
338  {
339 #if defined(ASIO_HAS_SIGNAL) || defined(ASIO_HAS_SIGACTION)
340  // Set signal handler back to the default if we're the last.
341  if (state->registration_count_[signal_number] == 1)
342  {
343 # if defined(ASIO_HAS_SIGACTION)
344  using namespace std; // For memset.
345  struct sigaction sa;
346  memset(&sa, 0, sizeof(sa));
347  sa.sa_handler = SIG_DFL;
348  if (::sigaction(signal_number, &sa, 0) == -1)
349 # else // defined(ASIO_HAS_SIGACTION)
350  if (::signal(signal_number, SIG_DFL) == SIG_ERR)
351 # endif // defined(ASIO_HAS_SIGACTION)
352  {
353 # if defined(ASIO_WINDOWS) || defined(__CYGWIN__)
354  ec = asio::error::invalid_argument;
355 # else // defined(ASIO_WINDOWS) || defined(__CYGWIN__)
356  ec = asio::error_code(errno,
357  asio::error::get_system_category());
358 # endif // defined(ASIO_WINDOWS) || defined(__CYGWIN__)
359  return ec;
360  }
361  }
362 #endif // defined(ASIO_HAS_SIGNAL) || defined(ASIO_HAS_SIGACTION)
363 
364  // Remove the registration from the set.
365  *deletion_point = reg->next_in_set_;
366 
367  // Remove the registration from the registration table.
368  if (registrations_[signal_number] == reg)
369  registrations_[signal_number] = reg->next_in_table_;
370  if (reg->prev_in_table_)
371  reg->prev_in_table_->next_in_table_ = reg->next_in_table_;
372  if (reg->next_in_table_)
373  reg->next_in_table_->prev_in_table_ = reg->prev_in_table_;
374 
375  --state->registration_count_[signal_number];
376 
377  delete reg;
378  }
379 
380  ec = asio::error_code();
381  return ec;

529 {
530  signal_state* state = get_signal_state();
531  static_mutex::scoped_lock lock(state->mutex_);
532 
533  if (service->next_ || service->prev_ || state->service_list_ == service)
534  {
535 #if !defined(ASIO_WINDOWS) \

539  int read_descriptor = state->read_descriptor_;
540  lock.unlock();
541  service->reactor_.deregister_descriptor(
542  read_descriptor, service->reactor_data_, false);
543  lock.lock();
544 #endif // !defined(ASIO_WINDOWS)
545  // && !defined(ASIO_WINDOWS_RUNTIME)
546  // && !defined(__CYGWIN__)
547

549  if (state->service_list_ == service)
550  state->service_list_ = service->next_;
551  if (service->prev_)
552  service->prev_->next_ = service->next_;
553  if (service->next_)
554  service->next_->prev_= service->prev_;
555  service->next_ = 0;
556  service->prev_ = 0;
557 
558 #if !defined(ASIO_WINDOWS) && !defined(__CYGWIN__)
559  // If this is the last service to be removed, close the pipe.
560  if (state->service_list_ == 0)
561  close_descriptors();
